CREATE TABLE Person(
	ID NUMBER NOT NULL,
	Email VARCHAR2(20) UNIQUE NOT NULL,
	Password VARCHAR2(30) NOT NULL,
	PRIMARY KEY(ID)
); 

CREATE TABLE Admin(
	AdminID NUMBER NOT NULL,
	FOREIGN KEY(AdminID) references Person
);

CREATE TABLE Member(
	MemberID  NUMBER NOT NULL,
	Wallet NUMBER DEFAULT 500,
	FOREIGN KEY(MemberID) references Person
);
